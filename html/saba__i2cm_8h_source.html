169 
170  switch(status)
171  {
172  case TW_START:
173  case TW_REP_START:
174  if( cmd.bytesToWrite == bytesWritten )
175  twdr= cmd.address | BIT(0);
176  else
177  twdr= cmd.address;
178 
179  cr= BIT(TWINT) | BIT(TWEN);
180  break;
181 
182  case TW_MT_SLA_ACK:
183  if( bytesWritten >= cmd.bytesToWrite )
184  {
185  cr= BIT(TWINT) | BIT(TWSTO) | BIT(TWEN);
186  }
187  else
188  {
189  twdr= cmd.writeBuffer[bytesWritten++];
190  cr= BIT(TWINT) | BIT(TWEN);
191  }
192  break;
193 
194  case TW_MR_SLA_ACK:
195  if( bytesRead >= cmd.bytesToRead )
196  cr= BIT(TWINT) | BIT(TWEN);
197  else
198  cr= BIT(TWINT) | BIT(TWEN) | BIT(TWEA);
199  break;
200 
201  case TW_MT_DATA_ACK:
202  if( bytesWritten >= cmd.bytesToWrite )
203  {
204  if( cmd.bytesToRead > 0)
205  cr= BIT(TWINT) | BIT(TWSTA) | BIT(TWEN);
206  else

208  }
209  else
